###

```jsx
const API_KEY = 'some-secret-api';

useEffect(() => {
  fetch(`https://www.test.com/api/dummy.json?apiKey=${API_KEY}`)
  // ...
}, [])
```

Of course we don't want it to be hardcoded.

```jsx
const API_KEY = proccess.env.NEXT_PUBLIC_API_KEY;

useEffect(() => {
  fetch(`https://www.test.com/api/dummy.json?apiKey=${API_KEY}`)
  // ...
}, [])
```

```js
export default async function handler(req, res) {
  const data = await fetch(
    `https://www.test.com/api/dummy.json?apiKey=${process.env.API_KEY}`,
  ).then(response => response.json());

  res.json(data);
}
```
