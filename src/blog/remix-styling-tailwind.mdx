### Dependencies <a name="dependencies"></a>

- [`autoprefixer`](https://github.com/postcss/autoprefixer)
- [`postcss`](https://github.com/postcss/postcss)
- [`postcss-cli`](https://github.com/postcss/postcss-cli)
- [`postcss-import`](https://github.com/postcss/postcss-import)
- [`tailwindcss`](https://github.com/tailwindlabs/tailwindcss)

```sh
npm install -D autoprefixer postcss postcss-cli postcss-import tailwindcss

yarn add -D autoprefixer postcss postcss-cli postcss-import tailwindcss
```

```json package
"scripts": {
  // ...
  "build": "yarn css:build && remix build",
  "css:build": "postcss styles/**/*.css -d app/styles",
  "css:watch": "yarn css:build --w",
  // ...
}
```

```css fileName=styles/tailwind
@tailwind base;
@tailwind components;
@tailwind utilities;
@tailwind screens;
```

```css fileName=styles/app
:root {
  --color-primary-light: hsl(210, 100%, 98%);
  --color-primary-100: hsl(210, 100%, 95%);
  --color-primary-200: hsl(210, 100%, 85%);
```

```js fileName=postcss.config
module.exports = {
  plugins: [
    require("tailwindcss"),
    require("autoprefixer"),
    require("postcss-import"),
  ],
};
```

```js fileName=tailwind.config
module.exports = {
  mode: process.env.NODE_ENV ? "jit" : undefined,
  // To purge CSS in .ts .tsx files
  purge: ["./app/**/*.{ts,tsx}"], 
  darkMode: "media", // Use media queries for dark mode
  theme: {
    extend: {
      colors: {
        // color scheme is defined in /app.css
        primary: {
          light: "var(--color-primary-light)",
          100: "var(--color-primary-100)",
          200: "var(--color-primary-200)",
          300: "var(--color-primary-300)",
```

```tsx fileName=app/root
// ...
import type { LinksFunction } from "remix";
import tailwindStyles from "~/styles/tailwind.css";
import appStyles from "~/styles/app.css";

export let links: LinksFunction = () => {
  return [
    { rel: "stylesheet", href: tailwindStyles },
    {
      rel: "stylesheet",
      href: appStyles,
    },
  ];
};
// ...
```
